('0' + d) : d; var h =date.getHours(); h = h < 10 ? ('0' + h) : h; var M =date.getMinutes(); M = M < 10 ? ('0' + M) : M; var s =date.getSeconds(); s = s < 10 ? ('0' + s) : s; return y + '-' + m + '-' + d + ' ' + h + ':' + M + ':' + s; } formatDate(new Date());
